function yn = filterSystem(xn, wn, N, T, dwt_mode)
    N = length(xn);
    M = size(wn, 2);
    L = fix(M / 2); % extend size // 2

    if dwt_mode == 1
        yn = zeros(1, N); % result
        yn = cast(yn, 'like', xn);

        yn(i) = wn * x(1, i:i + M - 1)';
    end
